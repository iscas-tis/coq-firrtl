Given $tmap$, the semantics of a statement $s$ in $M$ is defined by a relation $(vm, ct) \xrightarrow[tmap]{s} (vm', ct')$. %, where $s$ is seen as a modifier of module graphs. 
Moreover, the relation can be easily extended to sequences of statements, that is, $(vm, ct) \xrightarrow[tmap]{ss} (vm', ct')$, where $ss$ is a sequence of statements. 

In the sequel, we present the rules for defining the semantics of statements. Note that we only present the rules for the normal situations and ignore those rules for the abnormal ones, to avoid tediousness. 
       
%Let $M$ be a {\hifirrtl} module and $G$ be a module graph. We define how $G = (vm, ct)$ \emph{matches} $M$ as follows: We define the relations $(vm, ct) \xrightarrow[tmap]{s} (vm', ct')$, where $s$ is a statement. Then $G$ matches $M$ if $(vm_{emp}, ct_{emp}) \xrightarrow[tmap]{ss} (vm, ct)$, where $ss$ is the statement sequence of $M$.

    \infer[update\_vm]
    {update\_vm(vm, id, ft) = vm'}
    {
    \begin{array}{r l l}
    \multicolumn{3}{c}{id = (v, m) \wedge size\_of\_ftype(ft) = sz \wedge list\_gtype(ft) = ls} \\
    \forall n.\ 0 \le n < sz & \Rightarrow & vm'(v, m+n) = ls[n] \\
    \forall v', n'.\ (v' \neq v \vee n' < m \vee n' \ge m+sz) & \Rightarrow & vm'(v', n') = vm(v', n')  \\
    \end{array}
    }
    \\[1ex]
    \infer[update\_ct]
    {update\_ct(ct, id, ft, ls) = ct'}
    {
    \begin{array}{r l l}
    \multicolumn{3}{c}{id = (v, m) \wedge size\_of\_ftype(ft) = sz} \\
    \forall n.\ 0 \le n < sz & \Rightarrow & ct'(v, m+n) = ls[n] \\
    \forall v', n.\ (v' \neq v \vee n' < m \vee n' \ge m+sz) & \Rightarrow & ct(v', n') = ct'(v', n') \\
    \end{array}
    }
    \\[1ex]
    \infer[update\_ct\_ref]
    {update\_ct\_ref(ct, id, ft, ls, id', ft', ls', ct')}
    {
    \begin{array}{c}
    id = (v, m) \wedge id' = (v', m') \wedge size\_of\_ftype(ft) = sz\ \wedge \\
    ls\_gtype\_ref(ft', false) = lgt \\
    \forall n, ori'.\ 0 \le n < sz \wedge lgt[n] = (\_, false) \Rightarrow \\
    \left(
    \begin{array}{l}
    ct(v, m+n) \neq none \wedge ct'(v, m+n) = ls'[n]\ \wedge  \\
    ((v',m') \neq (v, m)  \rightarrow ct'(v', m' + n) = ct(v', m'+n)) 
    \end{array}
    \right)
    \\ 
    \forall n.\ 0 \le n < sz \wedge lgt[n] = (\_, true) \Rightarrow \\
    \left(
    \begin{array}{l}
    ct(v', m'+n) \neq none \wedge ct'(v', m'+n) = ls[n]\ \wedge  \\
    ((v, m) \neq (v', m')  \rightarrow ct'(v, m + n) = ct(v, m+n))
    \end{array}
    \right)
     \\
     \forall v'', n''.\ 
     \left(
     \begin{array}{l}
     (v'' \neq v \vee n'' < m \vee n'' \ge m+sz)\ \wedge \\
     (v'' \neq v' \vee n'' < m' \vee n'' \ge m'+sz)
     \end{array}
     \right)
       \Rightarrow \\
    ct'(v'', n'') = ct(v'', n'') 
    \end{array}
    }
    \\[1ex]
    \infer[invalidate]
    {invalidate(ct, id, ft, ori) = ct'}
    {
    \begin{array}{c}
    id = (v, m) \wedge size\_of\_ftype(ft) = sz \wedge ls\_gtype\_ref(ft, false) = lgt \\
    \forall n.\ 0 \le n < sz \wedge 
    \left(
    \begin{array}{l}
    (lgt[n] = (\_, false) \wedge (ori = \sink \vee ori = \duplex))\ \vee \\
    (lgt[n] = (\_, true) \wedge (ori = \source \vee ori = \duplex))
     \end{array}
    \right) \Rightarrow\\
    \hspace{1cm}  ct(v, m+n) \neq none \wedge ct'(v, m+n) = invalidated  \\
    \forall n.\ 0 \le n < sz \wedge 
    \left(
    \begin{array}{l}
    (lgt[n] = (\_, false) \wedge ori = \source)\ \vee \\
     (lgt[n] = (\_, true) \wedge ori = \sink)
     \end{array}
    \right)  
    \Rightarrow\\
    \hspace{1cm}  ct'(v, m+n) = ct(v, m+n)  \\
    \forall v', n'.\ (v' \neq v \vee n' < m  \vee n' \ge m+sz)  \rightarrow ct'(v', n') = ct(v', n')
    \end{array}
    }
  \end{gather*}

    {type\_compatible(ft, ft', flipped)}
    {
    \begin{array}{c}
    	ft = (ground, t) \wedge ft' = (ground, t) \\
	 t \in \{UInt, SInt\} \times \Nat \cup \{Clock, Reset, AsyncReset\}
    \end{array}
    }

    {type\_compatible(ft, ft', flipped)}
    {
    \begin{array}{c}
    	ft = (ground, t) \wedge ft' = (ground, t') \\
	(t = (UIntImp, n) \wedge t' = (UInt, n')) \vee (t = (SIntImp, n) \wedge t' = (SInt, n')) \\
	flipped = false \rightarrow n \ge n'
    \end{array}
    }
    \\[1ex]
    \infer[type\_compatible: imp-gt-2]
    {type\_compatible(ft, ft', flipped)}
    {
    \begin{array}{c}
    	ft = (ground, t) \wedge ft' = (ground, t') \\
	(t = (UInt, n) \wedge t' = (UIntImp, n')) \vee (t = (SInt, n) \wedge t' = (SIntImp, n')) \\
	flipped = true \rightarrow n \le n'
    \end{array}
    }
    \\[1ex]
    \infer[type\_compatible: imp-gt-3]
    {type\_compatible(ft, ft', flipped)}
    {
    \begin{array}{c}
    	ft = (ground, t) \wedge ft' = (ground, t') \\
	(t = (UIntImp, n) \wedge t' = (UIntImp, n')) \vee (t = (SIntImp, n) \wedge t' = (SIntImp, n')) \\
