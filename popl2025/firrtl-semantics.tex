\begin{definition}[Vertex]
	A vertex $V$ is a tuple $(IP, OP, \sort, \width, \datatype)$ where $IP$ is a set of input ports and OP is a set of output ports, ...
   \begin{itemize}
		\item {\sort} which assigns a sort to $v$; 		
		\item {\width} which assigns a nonnegative number to $v$;
		\item  {\datatype} which assigns a ground type to $v$
	\end{itemize}
\end{definition} 

\begin{definition}
	A hierarchical module graph is a tuple $\langle M_1, \cdots, M_n\rangle$ where each module $M_i$ has the following components:
	\begin{itemize}
		\item a finite set $V_i$ of vertices;
		\item a finite set $B_i$ of boxes; ($B_i$ and $V_i$ are all pairwise disjoint)
		\item a finite set $I_i$ of input ports;
		\item a finite set $O_i$ of output ports;
		\item a labeling function $Y_i: V_i \rightarrow \{i+1, \cdots, n\}$ that maps each box of the $i$-th graph to an index grater than $i$. That is, if $Y_i(b)=j$, for a box $b$ of graph $M_i$, then $b$ can be viewed as reference to the definition of the graph $M_j$. 
	\end{itemize}
\end{definition}
